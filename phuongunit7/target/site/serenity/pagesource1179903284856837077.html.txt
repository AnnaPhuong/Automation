function fieldTrack(ob){

		if(ob.value == ""){
			tracklink = ob.name+"?x=blank&amp;rkey="+Rkey;
		}else{
			tracklink = ob.name+"?x=filled&amp;rkey="+Rkey; 
		}

	}
